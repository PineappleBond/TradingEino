        agents := []string{

            "flow_analyzer",
            "position_manager",
            "sentiment_analyst",

        for _, agent := range agents {
            t.Run(agent+"_DESCRIPTION", func(t *testing.T) {
                path := fmt.Sprintf("internal/agent/%s/DESCRIPTION.md", agent)
                // Check file exists
                info, err := os.Stat(path)
                if err != nil {
                    t.Fatalf("DESCRIPTION.md not found for %s", agent)
                }
                // Check has content (>100 bytes)
                if info.Size() < 100 {
                    t.Fatalf("DESCRIPTION.md too small (%d bytes)", info.Size())
                }
            })

            t.Run(agent+"_SOUL", func(t *testing.T) {
                path := fmt.Sprintf("internal/agent/%s/SOUL.md", agent)
                info, err := os.Stat(path)
                if err != nil {
                    t.Fatalf("SOUL.md not found for %s", agent)
                }
                if info.Size() < 100 {
                    t.Fatalf("SOUL.md too small (%d bytes)", info.Size())
                }
            })
        }
    }
